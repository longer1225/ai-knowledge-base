# ---------------------- 工具函数 ----------------------
#用来发送http请求
def request_api(url, method="GET", data=None, files=None):
    try:
        if method == "GET":
            res = requests.get(url, params=data, timeout=30)
        else:
            if files:
                res = requests.post(url, files=files, timeout=60)
            else:
                res = requests.post(url, json=data, timeout=30)
        return res.json() if res.status_code == 200 else None
    except requests.exceptions.ConnectionError:
        st.error("❌ 后端服务未启动，请先运行 FastAPI 后端！")
    except Exception as e:
        st.error(f"❌ 请求错误：{str(e)}")
    return None

# ---------------------- 页面: 文档上传 ----------------------
def render_upload():
    st.markdown("<h2 style='color: white; margin-bottom: 20px;'>📤 文档上传</h2>", unsafe_allow_html=True)
    st.markdown("<p style='color: #aaa;'>支持 docx / txt 格式，单次限传1个（≤10MB）</p>", unsafe_allow_html=True)

    file = st.file_uploader("选择文件", type=["docx", "txt"], label_visibility="collapsed")
    if file:
        if st.button("开始上传与处理", type="primary"):
            with st.spinner("上传并处理文档中，请稍候..."):
                files = {"file": (file.name, file, file.type)}
                res = request_api(f"{API_BASE_URL}/api/upload", "POST", files=files)
                if res and res.get("code") == 0:
                    st.session_state.docs.append({
                        "id": res["data"]["doc_id"],
                        "name": file.name,
                        "size": file.size,
                        "time": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
                    })
                    st.success(f"✅ {file.name} 上传成功！")
                elif res:
                    st.error(res["msg"])

    st.markdown("<br>", unsafe_allow_html=True)

        for doc in st.session_state.docs:
            st.markdown(f"""
                <div class="dark-card">
                    <div style="display: flex; justify-content: space-between; align-items: center;">
                        <div>
                            <h4 style="margin:0; color:#ECECEC;">📄 {doc['name']}</h4>
                            <p style="margin:5px 0 0 0; color:#888; font-size:14px;">
                                大小: {doc['size']} B &nbsp;|&nbsp; 上传时间: {doc['time']}
                            </p>
                        </div>
                    </div>
                </div>
            """, unsafe_allow_html=True)
            # 删除按钮需要独立渲染以保持交互性
            if st.button("删除此文档", key=f"del_{doc['id']}"):
                st.session_state.docs.remove(doc)
                st.rerun()
    else:
        st.info("当前知识库为空，请前往「文档上传」页面添加文档。")
